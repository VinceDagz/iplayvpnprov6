local restrictedExecutors = {
    "Delta", "Delta Exploit", "DeltaExecutor"
}

local function getExecutorName()
    if identifyexecutor then
        return identifyexecutor()
    elseif getexecutorname then
        return getexecutorname()
    end
    return "Unknown"
end

local executorName = getExecutorName()

local isRestricted = false
for _, name in ipairs(restrictedExecutors) do
    if string.lower(executorName) == string.lower(name) then
        isRestricted = true
        break
    end
end

if isRestricted then
    game.Players.LocalPlayer:Kick("This script doesn't work on Delta Executor. Use other scripts like Krnl, Codex, ArceusX, Hydrogen for more better performance - Yukii")
else
    loadstring(game:HttpGet("https://raw.githubusercontent.com/VinceDagz/iplayvpnprov6/refs/heads/main/newupdategrowagarden.lua", true))()
end
